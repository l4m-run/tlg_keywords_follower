# Правила пересылки сообщений
#
# Формат: название_правила: ключевые_слова -> chat_id "Название группы" [case:on]

# Примеры:

# Срочные сообщения (без учёта регистра)
urgent: срочно, важно, breaking -> -1001234567890 "Важные новости"

# Торговые сигналы (несколько получателей)
trading: deal, price, buy -> -1005555555555 "Сигналы", -1006666666666 "Архив"

# Админские уведомления (с учётом регистра - [case:on])
admin: @Admin, Critical -> -1007777777777 "Админы" [case:on]
